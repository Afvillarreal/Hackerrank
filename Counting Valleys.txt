static int countingValleys(int n, string s) {
        int level = 1;
        char letter;
        int count = 0;

        for (int i = 0; i<n; i++){
            letter = s[i];

            if (level==0){
                count++;
            }
            
            if (letter== 'D'){
                level--;
            }
            if (letter== 'U'){
                level++;
            }
            
            //Console.WriteLine(s[i]);
        }
 
        return 0;
    }
